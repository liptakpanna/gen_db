CREATE TABLE public.variant_call
(
    vcid integer NOT NULL,
    chrom varchar(20),
    pos integer,
    CONSTRAINT variant_call_pkey PRIMARY KEY (vcid)
);

CREATE TABLE public.mutations
(
    mid integer NOT NULL,
    ref varchar(50),
    alt varchar(50),
    CONSTRAINT mutations_pkey PRIMARY KEY (mid)
);

CREATE TABLE public.samples
(
    sid integer NOT NULL,
    name varchar(20)
    CONSTRAINT mutations_pkey PRIMARY KEY (sid)
);

CREATE TABLE public.mapping
(
    sid integer,
    vcid integer,
    mid integer,
    quality real,
    coverage integer,
    allel_freq real, 
    CONSTRAINT map_mut_fk FOREIGN KEY (mid)
        REFERENCES public.mutations (mid) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION,
    CONSTRAINT map_var_fk FOREIGN KEY (vcid)
        REFERENCES public.variant_call (vcid) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION,
    CONSTRAINT map_sample_fk FOREIGN KEY (sid)
        REFERENCES public.sample (sid) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
);

CREATE TABLE public.sample_coverage
(
    sid integer,
    vcid integer,
    ref varchar(20),
    coverage integer, 
    CONSTRAINT cov_var_fk FOREIGN KEY (vcid)
        REFERENCES public.variant_call (vcid) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION,
    CONSTRAINT cov_sample_fk FOREIGN KEY (sid)
        REFERENCES public.sample (sid) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
);
